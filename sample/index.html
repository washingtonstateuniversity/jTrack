Running code


$(document).ready(function(){
    /**
    *  example of a json feed controlling the triggers
    *  since this is something that is cached this is a 
    *  super easy way to control may distrobutions with out
    *  having users that don't know, have to make edits
    **/
    $.getJSON('track/test.txt' , function(data){
        $.jtrack({
            load_analytics:{account:'UA-25040747-1'},
            trackevents:data // this can be hard codded here or set do be feed in like so
        });
    });
    
    //few other examples
    $('.normal a').track();
    $('.sidebar a').track({
        category : 'sidebar'
    });
    $('.complex a').track({
        category : function(element) { return element.attr('class').replace('tracked', '') }
    });

    //this sets the examples from doing anything but the google stuff
    $('a').click(function(e){
        e.preventDefault();
        e.stopPropagation();
    });
});
